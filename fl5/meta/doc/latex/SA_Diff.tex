\section{Description}
The eddy viscosity is expressed as
\begin{equation}
\varepsilon_m = \tilde{\nu_t} \, f_{\nu_1}
\end{equation}
The transport equation is written as (C\&C 6.3.14 and TC 6.3.10)
\begin{equation}
\begin{split}
u_x\frac{\partial \tilde{\nu_t}}{\partial x} + u_y \frac{\partial \tilde{\nu_t}}{\partial y} 
  = \;&c_{b_1} (1-f_{t_2}) \tilde{S} \tilde{\nu_t} \\
     &+\frac{1}{\sigma}   \bigg[\frac{\partial}{\partial y} \big( (\nu + \tilde{\nu_t})\frac{\partial \tilde{\nu_t}}{\partial y}   \big)  
                                                      +c_{b_2}  \big( \frac{\partial \tilde{\nu_t}}{\partial y} \big)^2                              \bigg]  \\
     &-(c_{w_1}f_w - \frac{c_{b_1}}{\kappa^2} f_{t_2}) \bigg(\frac{\tilde{\nu_t}}{d} \bigg)^2
\end{split}
\end{equation}
In the Inverse Boundary Layer (IBL) problem, the external velocity is an unknown variable which is computed as part of the BL solution. Using the transformation (TC 8.8.1)
\begin{equation} \label{eq_SATransport}
\begin{split}
\eta &= y\sqrt{\frac{u_\infty}{\nu x}}, \qquad y = \eta \sqrt{\frac{\nu \, x}{u_\infty}} \\
\psi (x,y) &= \sqrt{u_{\infty}\nu x} \;  f(x, \eta)
\end{split}
\end{equation}
The velocity components are expressed in terms of the stream function
$$u_x = \frac{\partial \psi}{\partial y} \qquad u_y = -\frac{\partial \psi}{\partial x}$$
Denote with a $^\prime$ the derivative w.r.t. $\eta$.
$$u_x = \frac{\partial }{\partial y} (\sqrt{u_{\infty}\nu x} \,  f) = \sqrt{u_{\infty}\nu x} \frac{\partial f}{\partial y} = u_\infty \frac{\partial f}{\partial \eta} =  u_\infty f^{\prime}$$
C\&C eq.4.5.7b:
\begin{equation}
\begin{split}
u_y &= -\sqrt{u_{\infty}\nu x} \big[\frac{f}{\sqrt{u_\infty x}} \frac{d}{dx} \sqrt{u_\infty x} + \frac{\partial f}{\partial x} +f^{\prime}\frac{\partial \eta}{\partial x} \big] \\
      &= -\sqrt{u_{\infty}\nu x} \big[\frac{f}{2x} + \frac{\partial f}{\partial x} +f^{\prime}\frac{\partial \eta}{\partial x} \big] 
\end{split}
\end{equation}


Divide eq. \eqref{eq_SATransport} by $\nu$ and define $\nu_t^+ = \chi = \tilde{\nu_t}/\nu$

Equation \eqref{eq_SATransport} becomes:
\begin{equation}
\begin{split}
u\frac{\partial \chi}{\partial x} + v \frac{\partial \chi}{\partial y} 
  = \;&c_{b_1} (1-f_{t_2}) \tilde{S} \chi \\
     &+\frac{1}{\sigma}   \bigg[\frac{\partial}{\partial y} \big( (\nu + \chi)\frac{\partial \chi}{\partial y}   \big)  
                                                      +c_{b_2}  \big( \frac{\partial \chi}{\partial y} \big)^2                              \bigg]  \\
     &-(c_{w_1}f_w - \frac{c_{b_1}}{\kappa^2} f_{t_2}) \bigg(\frac{\chi}{d} \bigg)^2
\end{split}
\end{equation}
