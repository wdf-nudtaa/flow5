\section{General}

Point vortex in the z direction 
$$\phi = -\frac{\gamma}{2\pi} \theta$$
$$\psi = \frac{\gamma}{2\pi} \log{r}$$
$$u_r = 0$$
$$u_{\theta} = -\frac{\gamma}{2 \pi r}$$
$$u(x,y) = u_r\cos{\theta} - u_{\theta} \sin{\theta}$$
$$v(x,y) = u_r\sin{\theta} + u_{\theta} \cos{\theta}$$
$$u(x,y) = \frac{\gamma}{2\pi} \frac{y}{x^2+y^2} $$
$$v(x,y) = -\frac{\gamma}{2\pi} \frac{x}{x^2+y^2} $$


\section{Linear vortex}
\includegraphics{stream}

Using barycentric coordinates (optional)
$$\gamma(x) = \gamma_1\lambda_1(x) + \gamma_2\lambda_2(x)$$
$$\lambda_1(x) = \frac{x_2-x}{x_2-x_1}$$
$$\lambda_2(x) = \frac{x-x_1}{x_2-x_1}$$
$$l = x_2-x_1$$

$$\gamma(x) = \frac{1}{l}(\gamma_1x_2-\gamma_2x_1) + \frac{x}{l}(\gamma_2-\gamma_1)$$

Constant part:
$$ C=\frac{1}{l}(\gamma_1x_2-\gamma_2x_1)$$

Linear part:
$$L(x)=\frac{x}{l}(\gamma_2-\gamma_1)$$

Polar coordinates
$$\cos{\theta}=\frac{X-x}{r}$$
$$r=\sqrt{(X-x)^2+Y^2}$$

\section{Stream function}
$$\psi(X,Y) = \int_{x_1}^{x_2} -\frac{\gamma(x)}{2\pi} \log{r} \,dx$$
$$\psi(X,Y) = -\frac{1}{2\pi l} \int_{x_1}^{x_2} \bigg( (\gamma_1x_2-\gamma_2x_1) + x(\gamma_2-\gamma_1)\bigg) \log{r} \,dx$$
$$\psi(X,Y) = -\frac{(\gamma_1x_2-\gamma_2x_1)}{2\pi l} \int_{x_1}^{x_2}  \log{r} \,dx  -\frac{(\gamma_2-\gamma_1)}{2\pi l} \int_{x_1}^{x_2} x \log{r} \,dx$$
$$\psi(X,Y) = \frac{(\gamma_1x_2-\gamma_2x_1)}{l}\psi_C(x) + \frac{(\gamma_2-\gamma_1)}{l}\psi_L(x)$$



Regroup terms in $\gamma_1$ and $\gamma_2$:
$$\psi(X,Y) = \frac{x_2 \psi_C - \psi_L}{l} \gamma_1 - \frac{x_1 \psi_C - \psi_L}{l} \gamma_2$$

Where:

Constant part:
$$\psi_C(X,Y) =  -\frac{1}{2\pi}\int_{x_1}^{x_2}  \log{r} \,dx = \frac{1}{2\pi} \big(X_2\log{r_2}-X_1\log{r_1} \big)$$

Linear part:
$$\psi_L(X,Y) = -\frac{1}{2\pi} \int_{x_1}^{x_2} x \log{r} \,dx = -\frac{1}{2\pi} \int_{x_1}^{x_2} x\log{\sqrt{(X-x)^2+Y^2}} \,dx $$
$$\psi_L(X,Y)=... $$

where
$$X_1 = X-x_1$$
$$X_2 = X-x_2$$
$$\beta_1 = \arctan\bigg(\frac{Y}{X-x_1}\bigg)$$
$$\beta_2 = \arctan\bigg(\frac{Y}{X-x_2}\bigg)$$


Note that $\psi_C$ is singular at the end points of the panel, but not $ \psi_L$  since $\lim(x\log(r))=0$.

\section{Velocity}
\subsection{u component of the velocity}
$$u(X,Y) = \frac{1}{2\pi} \int_{x_1}^{x_2}  \frac{\gamma(x) Y}{(X-x)^2+Y^2} \,dx$$
$$u(X,Y) = \frac{1}{2\pi} \int_{x_1}^{x_2}  \frac{(\gamma_1x_2-\gamma_2x_1) + x(\gamma_2-\gamma_1)}{l}\frac{Y}{(X-x)^2+Y^2} \,dx$$
$$u(X,Y) = \frac{(\gamma_1x_2-\gamma_2x_1)}{l}u_C(x) + \frac{(\gamma_2-\gamma_1)}{l}u_L(x)$$

with
$$u_C(X,Y) =  \frac{1}{2\pi}\int_{x_1}^{x_2}  \frac{Y}{(X-x)^2+Y^2} \,dx =  \frac{1}{2\pi}(\beta_2-\beta_1)$$
$$u_L(X,Y) =  \frac{1}{2\pi}\int_{x_1}^{x_2}  \frac{x Y}{(X-x)^2+Y^2} \,dx =  \frac{1}{2\pi}\bigg(Y(\log{r_2}-\log{r_1}) + X(\beta_2-\beta_1)\bigg)$$


\noindent
Regroup terms in $\gamma_1$ and $\gamma_2$:
$$u(X,Y) = \frac{x_2 u_C - u_L}{l} \gamma_1 - \frac{x_1 u_C - u_L}{l} \gamma_2$$


\noindent
\subsection{v component of the velocity}
$$v(X,Y) = -\frac{1}{2\pi} \int_{x_1}^{x_2}  \frac{\gamma(x) (X-x)}{(X-x)^2+Y^2} \,dx$$
$$v(X,Y) = -\frac{1}{2\pi} \int_{x_1}^{x_2}  \frac{(\gamma_1x_2-\gamma_2x_1) + x(\gamma_2-\gamma_1)}{l}\frac{X-x}{(X-x)^2+Y^2} \,dx$$
$$v(X,Y) = \frac{(\gamma_1x_2-\gamma_2x_1)}{l}v_C(x) + \frac{(\gamma_2-\gamma_1)}{l}v_L(x)$$

\noindent
with
$$v_C(X,Y) =  -\frac{1}{2\pi}\int_{x_1}^{x_2}  \frac{X-x}{(X-x)^2+Y^2} \,dx =  \frac{1}{2\pi}(\log{r_2}-\log{r_1})$$
$$v_L(X,Y) =  -\frac{1}{2\pi}\int_{x_1}^{x_2}  \frac{x(X-x)}{(X-x)^2+Y^2} \,dx $$
$$v_L(X,Y) =  -\frac{1}{2\pi}(X(\log{r_1}-\log{r_2}) +(x_1-x_2) +Y(\beta_2-\beta_1)$$


\noindent
Regroup terms in $\gamma_1$ and $\gamma_2$:
$$v(X,Y) = \frac{x_2 v_C - v_L}{l} \gamma_1 - \frac{x_1 v_C - v_L}{l} \gamma_2$$
